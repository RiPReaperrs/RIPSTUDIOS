import { clsx, type ClassValue } from "clsx"; import { twMerge } from "tailwind-merge"; export function cn(...inputs: ClassValue[]) { return twMerge(clsx(inputs)); } const SUFFIXES = ["", "K", "M", "B", "T", "Qa", "Qi", "Sx", "Sp", "Oc", "No", "Dc"]; export function formatNumber(num: number, decimals: number = 1): string { if (num < 1000) return Math.floor(num).toString(); const tier = Math.floor(Math.log10(num) / 3); if (tier >= SUFFIXES.length) return num.toExponential(2); const scale = Math.pow(10, tier * 3); const scaled = num / scale; if (scaled >= 100 || scaled % 1 === 0) return scaled.toFixed(0) + SUFFIXES[tier]; return scaled.toFixed(decimals) + SUFFIXES[tier]; } export function formatTime(seconds: number): string { const h = Math.floor(seconds / 3600); const m = Math.floor((seconds % 3600) / 60); const s = Math.floor(seconds % 60); if (h > 0) return `${h}h ${m}m ${s}s`; if (m > 0) return `${m}m ${s}s`; return `${s}s`; }
